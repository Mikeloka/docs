30  Window::Window(const GEngine::Application& gapp, const char* name, const WindowProperties& wp)
31  : window(), monitor(glfwGetPrimaryMonitor()), name(name), width(wp.width), height(wp.height),
32  fullscreen(wp.fullscreen), vsync(wp.vsync)
33 
34  {
35 
36  if (!glfwInit())
37  exit(EXIT_FAILURE);
38 
39  const GLFWvidmode* mode = glfwGetVideoMode(glfwGetPrimaryMonitor());
40 
41  if(fullscreen || vsync ||
42  width == 0 || height == 0)
43  {
44  glfwWindowHint(GLFW_RED_BITS, mode->redBits);
45  glfwWindowHint(GLFW_GREEN_BITS, mode->greenBits);
46  glfwWindowHint(GLFW_BLUE_BITS, mode->blueBits);
47  glfwWindowHint(GLFW_REFRESH_RATE, mode->refreshRate);
48  }
49 
50  if(width == 0 || height == 0) {
51  width = mode->width;
52  height = mode->height;
53  }
54 
55  if(fullscreen) {
56  if(vsync) {
57  window = glfwCreateWindow(width, height, name, glfwGetPrimaryMonitor(), NULL);
58  glfwSetWindowMonitor(window, glfwGetPrimaryMonitor(), 0, 0, width, height, mode->refreshRate);
59  }
60  else window = glfwCreateWindow(width, height, name, glfwGetPrimaryMonitor(), NULL);
61  } else {
62  if(vsync) {
63  window = glfwCreateWindow(width, height, name, NULL, NULL);
64  glfwSetWindowMonitor(window, NULL, 0, 0, width, height, mode->refreshRate);
65 
66  }
67  else window = glfwCreateWindow(width, height, name, NULL, NULL);
68  }
69 
70  if (!window)
71  {
72  glfwTerminate();
73  exit(EXIT_FAILURE);
74  }
75 
76  /* Make the window's context current */
77  glfwMakeContextCurrent(window);
78  if(gapp.gapi == GraphicsAPI::OpenGL) {
79  glfwSetWindowSizeCallback(window, [](GLFWwindow* window __attribute__((unused)), int width, int height) {
80  glViewport(0, 0, width, height);
81  });
82  }
83
